29pub(crate) fn start_block_processor_task<S, F>(
30    storage: S,
31    shutdown_signal: F,
32    limits: LimitsConfig,
33    options: NodeOptions,
34    block_exporter_id: u32,
35    destination_config: DestinationConfig,
36) -> (
37    UnboundedSender<BlockId>,
38    std::thread::JoinHandle<Result<(), ExporterError>>,
39)
40where
41    S: Storage + Clone + Send + Sync + 'static,
42    F: IntoFuture<Output = ()> + Clone + Send + Sync + 'static,
43    <F as IntoFuture>::IntoFuture: Future<Output = ()> + Send + Sync + 'static,
44{
45    let (task_sender, queue_front) = unbounded_channel();
46    let new_block_queue = NewBlockQueue {
47        queue_rear: task_sender.clone(),
48        queue_front,
49    };
50    let handle = std::thread::spawn(move || {
51        start_block_processor(
52            storage,
53            shutdown_signal,
54            limits,
55            options,
56            block_exporter_id,
57            new_block_queue,
58            destination_config,
59        )
60    });
61
62    (task_sender, handle)
63}
64
65struct NewBlockQueue {
66    pub(crate) queue_rear: UnboundedSender<BlockId>,
67    pub(crate) queue_front: UnboundedReceiver<BlockId>,
68}
69
70impl NewBlockQueue {
71    async fn recv(&mut self) -> Option<BlockId> {
72        let block = self.queue_front.recv().await;
73        #[cfg(with_metrics)]
74        crate::metrics::EXPORTER_NOTIFICATION_QUEUE_LENGTH.dec();
75        block
76    }
77
78    fn push_back(&self, block_id: BlockId) {
79        self.queue_rear
80            .send(block_id)
81            .expect("sender should never fail");
82        #[cfg(with_metrics)]
83        crate::metrics::EXPORTER_NOTIFICATION_QUEUE_LENGTH.inc();
84    }
85}

87#[tokio::main(flavor = "current_thread")]
88async fn start_block_processor<S, F>(
89    storage: S,
90    shutdown_signal: F,
91    limits: LimitsConfig,
92    options: NodeOptions,
93    block_exporter_id: u32,
94    new_block_queue: NewBlockQueue,
95    destination_config: DestinationConfig,
96) -> Result<(), ExporterError>
97where
98    S: Storage + Clone + Send + Sync + 'static,
99    F: IntoFuture<Output = ()> + Clone + Send + Sync + 'static,
100    <F as IntoFuture>::IntoFuture: Future<Output = ()> + Send + Sync + 'static,
101{
102    let destination_ids = destination_config
103        .destinations
104        .iter()
105        .map(|destination| destination.id())
106        .collect::<Vec<_>>();
107    let (block_processor_storage, mut exporter_storage) =
108        BlockProcessorStorage::load(storage.clone(), block_exporter_id, destination_ids, limits)
109            .await?;
110
111    let tracker = ExportersTracker::new(
112        options,
113        limits.work_queue_size.into(),
114        shutdown_signal.clone(),
115        exporter_storage.clone()?,
116        destination_config.destinations.clone(),
117    );
118
119    let mut block_processor = BlockProcessor::new(
120        tracker,
121        block_processor_storage,
122        new_block_queue,
123        destination_config.committee_destination,
124    );
125
126    block_processor
127        .run_with_shutdown(shutdown_signal, limits.persistence_period_ms)
128        .await?;
129
130    block_processor.pool_state().join_all().await;
131
132    Ok(())
133}

359    async fn test_committee_destination() -> anyhow::Result<()> {
360        tracing::info!("Starting test_committee_destination test");
361
362        let storage = DbStorage::<MemoryDatabase, _>::make_test_storage(None).await;
363        let test_chain = TestChain::new(storage.clone());
364        let cancellation_token = CancellationToken::new();
365        let child = cancellation_token.child_token();
366        let signal = ExporterCancellationSignal::new(child.clone());
367
368        let mut destinations = vec![];
369        let dummy_validator = spawn_dummy_validator(&mut destinations, &cancellation_token).await?;
370        let destination = destinations[0].clone();
371        let validator_state = ValidatorState {
372            network_address: destination.address(),
373            votes: 0,
374            account_public_key: AccountPublicKey::test_key(0),
375        };
376        let (notifier, block_processor_handle) = start_block_processor_task(
377            storage.clone(),
378            signal,
379            LimitsConfig {
380                persistence_period_ms: 3000,
381                ..Default::default()
382            },
383            NodeOptions {
384                send_timeout: Duration::from_millis(4000),
385                recv_timeout: Duration::from_millis(4000),
386                retry_delay: Duration::from_millis(1000),
387                max_retries: 10,
388            },
389            0,
390            DestinationConfig {
391                committee_destination: true,
392                destinations: vec![],
393            },
394        );
395
396        let mut single_validator = BTreeMap::new();
397        single_validator.insert(Secp256k1PublicKey::test_key(0), validator_state);
398
399        let confirmed_certificate = test_chain
400            .publish_committee(&single_validator, None)
401            .await
402            .expect("Failed to publish committee");
403
404        let first_notification = BlockId::from_confirmed_block(confirmed_certificate.value());
405        notifier.send(first_notification)?;
406        sleep(Duration::from_secs(4)).await;
407
408        {
409            let pinned = dummy_validator.state.pin();
410            assert!(pinned.contains(&first_notification.hash));
411        }
412        // We expect the validator to receive the confired certificate only once.
413        {
414            let pinned = dummy_validator.duplicate_blocks.pin();
415            assert!(pinned.get(&first_notification.hash).is_none());
416        }
417
418        ///////////
419        // Add new validator to the committee.
420        ///////////
421        let second_dummy = spawn_dummy_validator(&mut destinations, &cancellation_token).await?;
422        let destination = destinations[1].clone();
423        let validator_state = ValidatorState {
424            network_address: destination.address(),
425            votes: 0,
426            account_public_key: AccountPublicKey::test_key(1),
427        };
428        let mut two_validators = single_validator.clone();
429        two_validators.insert(Secp256k1PublicKey::test_key(1), validator_state);
430        let add_validator_certificate = test_chain
431            .publish_committee(&two_validators, Some(confirmed_certificate.value().clone()))
432            .await
433            .expect("Failed to publish new committee");
434        let second_notification = BlockId::from_confirmed_block(add_validator_certificate.value());
435        notifier.send(second_notification)?;
436        sleep(Duration::from_secs(4)).await;
437
438        {
439            let pinned = second_dummy.state.pin();
440            assert!(pinned.contains(&second_notification.hash));
441        }
442        // We expect the new validator to receive the new confirmed certificate only once.
443        {
444            let pinned = second_dummy.duplicate_blocks.pin();
445            assert!(pinned.get(&second_notification.hash).is_none());
446        }
447        // The first certificate should not be duplicated.
448        {
449            let pinned = second_dummy.duplicate_blocks.pin();
450            assert!(pinned.get(&first_notification.hash).is_none());
451        }
452
453        // The first validator should receive the new committee as well.
454        {
455            let pinned = dummy_validator.state.pin();
456            assert!(pinned.contains(&second_notification.hash));
457        }
458        // We expect the first validator to receive the new confirmed certificate only once.
459        {
460            let pinned = dummy_validator.duplicate_blocks.pin();
461            assert!(pinned.get(&second_notification.hash).is_none());
462        }
463
464        ///////////
465        // Remove the validator from the committee.
466        ///////////
467
468        let mut new_validators = two_validators.clone();
469        new_validators.remove(&Secp256k1PublicKey::test_key(0));
470
471        let remove_validator_certificate = test_chain
472            .publish_committee(
473                &new_validators,
474                Some(add_validator_certificate.value().clone()),
475            )
476            .await
477            .expect("Failed to publish new committee");
478
479        let third_notification =
480            BlockId::from_confirmed_block(remove_validator_certificate.value());
481        notifier.send(third_notification)?;
482        sleep(Duration::from_secs(4)).await;
483        // The first validator should not receive the new confirmed certificate.
484        {
485            let pinned = dummy_validator.state.pin();
486            assert!(!pinned.contains(&third_notification.hash));
487        }
488        // We expect the first validator to receive the new confirmed certificate only once.
489        {
490            let pinned = dummy_validator.duplicate_blocks.pin();
491            assert!(pinned.get(&third_notification.hash).is_none());
492        }
493        // The second validator should receive the new confirmed certificate.
494        {
495            let pinned = second_dummy.state.pin();
496            assert!(pinned.contains(&third_notification.hash));
497        }
498        // We expect the second validator to receive the new confirmed certificate only once.
499        {
500            let pinned = second_dummy.duplicate_blocks.pin();
501            assert!(pinned.get(&third_notification.hash).is_none());
502        }
503
504        cancellation_token.cancel();
505        block_processor_handle.join().unwrap()?;
506        Ok(())
507    }

509    struct TestChain<S> {
510        chain_description: ChainDescription,
511        storage: S,
512    }
513
514    impl<S> TestChain<S> {
515        fn new(storage: S) -> Self {
516            let chain_description = ChainDescription::new(
517                ChainOrigin::Root(0),
518                InitialChainConfig {
519                    ownership: Default::default(),
520                    epoch: Default::default(),
521                    balance: Default::default(),
522                    application_permissions: Default::default(),
523                    min_active_epoch: Epoch::ZERO,
524                    max_active_epoch: Epoch::ZERO,
525                },
526                Timestamp::now(),
527            );
528            Self {
529                chain_description,
530                storage,
531            }
532        }
533
534        // Constructs a new block, with the blob containing the committee.
535        async fn publish_committee(
536            &self,
537            validators: &BTreeMap<Secp256k1PublicKey, ValidatorState>,
538            prev_block: Option<ConfirmedBlock>,
539        ) -> Result<ConfirmedBlockCertificate, ViewError>
540        where
541            S: Storage + Clone + Send + Sync + 'static,
542        {
543            let committee = Committee::new(validators.clone(), ResourceControlPolicy::testnet());
544            let chain_id = self.chain_description.id();
545            let chain_blob = Blob::new_chain_description(&self.chain_description);
546
547            let committee_blob = Blob::new(BlobContent::new_committee(bcs::to_bytes(&committee)?));
548            let proposed_block = if let Some(parent_block) = prev_block {
549                make_child_block(&parent_block).with_operation(Operation::System(Box::new(
550                    SystemOperation::Admin(AdminOperation::CreateCommittee {
551                        epoch: parent_block.epoch().try_add_one().unwrap(),
552                        blob_hash: committee_blob.id().hash,
553                    }),
554                )))
555            } else {
556                make_first_block(chain_id).with_operation(Operation::System(Box::new(
557                    SystemOperation::Admin(AdminOperation::CreateCommittee {
558                        epoch: Epoch::ZERO,
559                        blob_hash: committee_blob.id().hash,
560                    }),
561                )))
562            };
563            let blobs = vec![chain_blob, committee_blob];
564            let block = BlockExecutionOutcome {
565                blobs: vec![blobs.clone()],
566                ..Default::default()
567            }
568            .with(proposed_block);
569
570            let confirmed_block = ConfirmedBlock::new(block);
571            let certificate = ConfirmedBlockCertificate::new(confirmed_block, Round::Fast, vec![]);
572            self.storage
573                .write_blobs_and_certificate(blobs.as_ref(), &certificate)
574                .await?;
575
576            Ok(certificate)
577        }
578    }
